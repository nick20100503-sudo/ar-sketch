    <script>

        const imageLoader = document.getElementById('image-loader');

        const opacitySlider = document.getElementById('opacity-slider');

        const scaleSlider = document.getElementById('scale-slider');

        const thresholdSlider = document.getElementById('threshold-slider');

        const filterBtn = document.getElementById('filter-btn');

        const thresholdRow = document.getElementById('threshold-row');

        const arImage = document.getElementById('ar-patch');

        const canvas = document.getElementById('hidden-canvas');

        const ctx = canvas.getContext('2d');

        let originalImgDataUrl = null;

        let currentImageObject = null;

        let isLineMode = false;

        // 監聽圖片上傳

        imageLoader.addEventListener('change', function(e) {

            const reader = new FileReader();

            reader.onload = function(event) {

                originalImgDataUrl = event.target.result;

               

                currentImageObject = new Image();

                currentImageObject.onload = function() {

                    // 調整 AR 顯示比例

                    const aspect = currentImageObject.width / currentImageObject.height;

                    if (aspect > 1) {

                        arImage.setAttribute('width', aspect);

                        arImage.setAttribute('height', 1);

                    } else {

                        arImage.setAttribute('width', 1);

                        arImage.setAttribute('height', 1 / aspect);

                    }

                   

                    // 根據目前模式渲染圖片

                    renderImage();

                };

                currentImageObject.src = originalImgDataUrl;

            };

            if(e.target.files[0]) {

                reader.readAsDataURL(e.target.files[0]);

            }

        });

        // 切換模式按鈕

        filterBtn.addEventListener('click', function() {

            if (!originalImgDataUrl) return alert("請先上傳圖片！");

            isLineMode = !isLineMode;

           

            if (isLineMode) {

                filterBtn.innerText = "線稿模式";

                filterBtn.classList.add('active');

                thresholdRow.style.display = 'flex';

            } else {

                filterBtn.innerText = "原圖模式";

                filterBtn.classList.remove('active');

                thresholdRow.style.display = 'none';

            }

            renderImage();

        });

        // 調整細節滑桿時即時重新計算線稿

        thresholdSlider.addEventListener('input', function() {

            if (isLineMode && currentImageObject) {

                renderImage();

            }

        });

        // 核心渲染邏輯

        function renderImage() {

            if (!currentImageObject) return;

            if (!isLineMode) {

                // 原圖模式：直接換上原本的 Base64

                arImage.setAttribute('src', originalImgDataUrl);

            } else {

                // 線稿模式：用 Canvas 進行 Sobel 邊緣檢測

                const w = currentImageObject.width;

                const h = currentImageObject.height;

                canvas.width = w;

                canvas.height = h;

               

                ctx.drawImage(currentImageObject, 0, 0, w, h);

                const imgData = ctx.getImageData(0, 0, w, h);

               

                // 執行邊緣檢測

                const edgeData = sobelFilter(imgData, parseInt(thresholdSlider.value));

                ctx.putImageData(edgeData, 0, 0);

               

                // 將 Canvas 轉成圖片傳給 AR 顯示

                arImage.setAttribute('src', canvas.toDataURL());

            }

        }

        // Sobel 邊緣檢測演算法 (提取輪廓線)

        function sobelFilter(imgData, threshold) {

            const w = imgData.width;

            const h = imgData.height;

            const src = imgData.data;

            const output = ctx.createImageData(w, h);

            const dst = output.data;

            // 先建立灰階矩陣

            const gray = new Uint8ClampedArray(w * h);

            for (let i = 0; i < src.length; i += 4) {

                // 經典灰階權重公式

                gray[i / 4] = 0.299 * src[i] + 0.587 * src[i + 1] + 0.114 * src[i + 2];

            }

            // 算運算子邊緣

            for (let y = 1; y < h - 1; y++) {

                for (let x = 1; x < w - 1; x++) {

                    const idx = y * w + x;

                    // Sobel 卷積核矩陣

                    const valX = (

                        -1 * gray[idx - w - 1] + 1 * gray[idx - w + 1] +

                        -2 * gray[idx - 1]     + 2 * gray[idx + 1] +

                        -1 * gray[idx + w - 1] + 1 * gray[idx + w + 1]

                    );

                    const valY = (

                        -1 * gray[idx - w - 1] - 2 * gray[idx - w] - 1 * gray[idx - w + 1] +

                        1 * gray[idx + w - 1] + 2 * gray[idx + w] + 1 * gray[idx + w + 1]

                    );

                    const mag = Math.sqrt(valX * valX + valY * valY);

                   

                    // 根據閥值判定是否為邊緣線條 (這裡做成白底黑線，最適合臨摹)

                    const pixelIdx = idx * 4;

                    if (mag > threshold) {

                        dst[pixelIdx] = 0;     // 紅

                        dst[pixelIdx + 1] = 0; // 綠

                        dst[pixelIdx + 2] = 0; // 藍

                        dst[pixelIdx + 3] = 255; // 不透明度

                    } else {

                        dst[pixelIdx] = 255;   // 白底

                        dst[pixelIdx + 1] = 255;

                        dst[pixelIdx + 2] = 255;

                        dst[pixelIdx + 3] = 0; // 將白底設為完全透明，這樣才能看到底下的紙！

                    }

                }

            }

            return output;

        }

        scaleSlider.addEventListener('input', e => arImage.setAttribute('scale', `${e.target.value} ${e.target.value} 1`));
